SDK & Documentation

Integrate NyxAuth into your application in minutes. Download a header-only, dependency-light SDK for your language and follow the quick-start guide below.

Quick Start
Program.cs 
// 1. Add NyxAuth.cs to your project
// 2. Install Newtonsoft.Json via NuGet
// 3. Replace with your app's secret key

using NyxAuthSDK;

var nyx = new NyxAuth("YOUR_SECRET_KEY");

// Login with username + password
var result = await nyx.Login("username", "password");
if (result.Success)
{
    Console.WriteLine("Welcome, " + result.Data["username"]);
    // Grant access to your app
}
else
{
    Console.WriteLine("Error: " + result.Error);
}

// Login with HWID locking
string hwid = Environment.GetEnvironmentVariable("COMPUTERNAME");
var locked = await nyx.Login("username", "password", hwid);

// Validate a license key (no username)
var lic = await nyx.CheckLicense("NYX-XXXX-XXXX-XXXX");
if (lic.Success) Console.WriteLine("License valid!");
main.cpp 
// 1. Add nyxauth.hpp to your project
// 2. Link with libcurl and nlohmann/json
// 3. Replace with your app's secret key

#include "nyxauth.hpp"

int main() {
    NyxAuthSDK::NyxAuth nyx("YOUR_SECRET_KEY");

    // Login with username + password
    auto result = nyx.Login("username", "password");
    if (result.success) {
        std::cout << "Welcome!" << std::endl;
    }

    // Login with HWID lock
    auto locked = nyx.Login("username", "password", GetHWID());

    // Validate a license key
    auto lic = nyx.CheckLicense("NYX-XXXX-XXXX-XXXX");
    if (lic.success) std::cout << "License valid!" << std::endl;

    return 0;
}
main.py 
# 1. Add nyxauth.py to your project
# 2. pip install requests
# 3. Replace with your app's secret key

from nyxauth import NyxAuth

nyx = NyxAuth("YOUR_SECRET_KEY")

# Login with username + password
result = nyx.login("username", "password")
if result["success"]:
    print(f"Welcome, {result['user']['username']}")

# Login with HWID lock
import platform
result = nyx.login("username", "password", hwid=platform.node())

# Validate a license key
lic = nyx.check_license("NYX-XXXX-XXXX-XXXX")
if lic["success"]: print("License valid!")
