Quick Start
Program.cs
// 1. Add AuthForge.cs to your project
// 2. Install Newtonsoft.Json via NuGet
// 3. Replace with your app's secret key

using AuthForgeSDK;

var af = new AuthForge("YOUR_SECRET_KEY");

// Login with username + password
var login = await af.Login("username", "password");
if (login.Success)
{
    Console.WriteLine("Welcome, " + login.Data["username"]);
    // Grant access to your app
}
else
{
    Console.WriteLine("Error: " + login.Error);
    // Block access
}

// Login with HWID lock enabled
string hwid = Environment.GetEnvironmentVariable("COMPUTERNAME");
var loginHwid = await af.Login("username", "password", hwid);

// Validate a license key only (no username needed)
var check = await af.CheckLicense("LICENSE-KEY");
if (check.Success) Console.WriteLine("License valid!");
main.cpp
// 1. Add authforge.hpp to your project
// 2. Install libcurl and nlohmann/json
// 3. Replace with your app's secret key

#include "authforge.hpp"

int main() {
    AuthForge af("YOUR_SECRET_KEY");

    // Login with username + password
    auto login = af.Login("username", "password");
    if (login.success) {
        std::cout << "Welcome, " << login.message << std::endl;
    } else {
        std::cout << "Error: " << login.error << std::endl;
    }

    // Login with HWID
    auto loginHwid = af.Login("username", "password", GetHWID());

    // Check a license key
    auto check = af.CheckLicense("LICENSE-KEY");
    if (check.success) std::cout << "License valid!" << std::endl;

    return 0;
}
main.py
# 1. Add authforge.py to your project
# 2. pip install requests
# 3. Replace with your app's secret key

from authforge import AuthForge

af = AuthForge("YOUR_SECRET_KEY")

# Login with username + password
login = af.login("username", "password")
if login["success"]:
    print(f"Welcome, {login['user']['username']}")
else:
    print(f"Error: {login['message']}")

# Login with HWID
import platform
hwid = platform.node()
login = af.login("username", "password", hwid=hwid)

# Check a license key
check = af.check_license("LICENSE-KEY")
if check["success"]: print("License valid!")
